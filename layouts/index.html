{{ define "main" }} {{- $user := "me" -}} {{- $host := "jacobcolvin.com" -}} {{- $headerLine := printf "%s@%s" $user $host -}} {{- $birth := time.AsTime (printf "%sT00:00:00Z" .Site.Params.author.born) -}} {{- $tnow := now -}} {{- $bdThis := time.AsTime (printf "%04d-10-03T00:00:00Z" $tnow.Year) -}} {{- $years := sub $tnow.Year $birth.Year -}} {{- $bdLast := $bdThis -}} {{- if lt $tnow.Unix $bdThis.Unix -}} {{- $years = sub $years 1 -}} {{- $bdLast = time.AsTime (printf "%04d-10-03T00:00:00Z" (sub $tnow.Year 1)) -}} {{- end -}} {{- $days := div (sub $tnow.Unix $bdLast.Unix) 86400 -}} {{- $uptime := printf "%d years, %d days" $years $days -}}
{{- /* Mask codes (O Y G B P C R M, space = no color) → .aod-X in main.scss. */ -}} {{- $ascii := readFile "layouts/partials/page/home-ascii.txt" -}} {{- $colors := readFile "layouts/partials/page/home-colors.txt" -}} {{- $artLines := split (trim $ascii "\n") "\n" -}} {{- $clrLines := split (trim $colors "\n") "\n" -}} 
  {{- range $r, $line := $artLines -}}
    {{- if ne $r 0 }}{{ "\n" }}{{ end -}}
    {{- $mask := index $clrLines $r -}}
    {{- $runColor := "" -}}{{- $runText := "" -}}
    {{- range $col, $cc := split $line "" -}}
      {{- $m := substr $mask $col 1 -}}
      {{- $color := cond (or (eq $m " ") (eq $m "")) "" $m -}}
      {{- if eq $color $runColor -}}
        {{- $runText = printf "%s%s" $runText $cc -}}
      {{- else -}}
        {{- if eq $runColor "" -}}{{ $runText -}}
        {{- else -}}{{ $runText }}{{- end -}}
        {{- $runColor = $color -}}{{- $runText = $cc -}}
      {{- end -}}
    {{- end -}}
    {{- if eq $runColor "" -}}{{ $runText -}}
    {{- else -}}{{ $runText }}{{- end -}}
  {{- end -}}
